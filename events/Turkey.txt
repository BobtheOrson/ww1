#Ottoman Civil War
country_event = {
	id = turkey.11 
	title = turkey.11.t 
	desc = turkey.11.d 
	picture = GFX_report_event_spain_civil_war_soldiers
	
	trigger = {
		OR = {
			has_country_flag = TUR_monarchy_1_flag
			has_country_flag = TUR_monarchy_2_flag
			has_country_flag = TUR_monarchy_3_flag
			has_country_flag = TUR_monarchy_4_flag
		}
		date > 1915.1.1 
	}
	
	fire_only_once = yes 
	
	mean_time_to_happen = {
		days = 90 
	}
	
	option = {
		name = turkey.11.a 
		hidden_effect = {
			set_capital = 341
			start_civil_war = {
				ruling_party = fascism
				ideology = neutrality
				size = 0.5
				capital = 49
				states = { 49 343 346 348 349 350 353 352 680 676 291 675 455 679 293 }
			}
			random_country = {
				limit = {
					original_tag = TUR
					has_government = neutrality
				}
				
				set_national_unity = 0.6
				add_popularity = {
					ideology = fascism
					popularity = 0.37
				}
			}
			news_event = { hours = 12 id = news.243 }
			set_global_flag = ottoman_civil_war
			
			set_national_unity = 0.8
			
			add_popularity = {
				ideology = fascism
				popularity = 1
			}
		}
		custom_effect_tooltip = TUR_civil_war_begins_tt
	}
}
